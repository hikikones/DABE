%----------------------------------------------------------------------------------------
%	PACKAGES AND OTHER DOCUMENT CONFIGURATIONS
%----------------------------------------------------------------------------------------

\documentclass[11pt]{report}

% New lines, no indent
\usepackage[parfill]{parskip}

% Encoding of international characters
\usepackage[utf8]{inputenc}
\usepackage[english]{babel}
\usepackage{csquotes}

% Hyperlinks
\usepackage{hyperref}
\usepackage[dvipsnames]{xcolor}
\hypersetup{
    colorlinks=true,
    linkcolor={NavyBlue},
	urlcolor={NavyBlue},
	citecolor={Maroon}
}

% Palatino font
\usepackage{mathpazo}

% Image support
\usepackage{graphicx, float}
\usepackage[font=small,labelfont=bf]{caption}
\graphicspath{{images/}}

% Syntax highlighting
\usepackage{listings}
\definecolor{LightGray}{gray}{0.95}
\lstset{
	backgroundcolor=\color{LightGray},
	commentstyle=\color{OliveGreen},
	rulecolor=\color{Black},
	stringstyle=\color{BrickRed},
	basicstyle=\footnotesize,
	language=C,
	columns=fullflexible,
	frame=tlbr,
	framesep=5pt,
	framerule=0pt,
	captionpos=b,
	breakatwhitespace=false,
	mathescape=true,
	showspaces=false,
	showstringspaces=false,
	numbers=left,
	numbersep=10pt,
	showtabs=false,
	tabsize=2,
	breaklines=true
}

% Glossaries, acronyms etc
\usepackage[nopostdot,toc,acronym,nomain,nonumberlist]{glossaries}
\makeglossaries
\loadglsentries{metadata/terms}
\renewcommand*{\glstextformat}[1]{\textcolor{Brown}{#1}}

% Bibliography
\usepackage[backend=biber,style=alphabetic]{biblatex}
\addbibresource{metadata/refs.bib}
